function parseAndRender(text) { if (!text) return ''; let html = ''; // Separa blocos de código ```...``` const parts = text.split(/(```[\s\S]*?```)/g); for (const part of parts) { // BLOCO DE CÓDIGO if (part.startsWith('```') && part.endsWith('```')) { const content = part.slice(3, -3); let lang = 'CODE'; let code = content; const firstLineBreak = content.indexOf('\n'); if (firstLineBreak > 0) { const possibleLang = content.substring(0, firstLineBreak).trim(); if (/^[a-zA-Z0-9_-]+$/.test(possibleLang)) { lang = possibleLang.toUpperCase(); code = content.substring(firstLineBreak + 1); } } const blockId = 'cb_' + Math.random().toString(36).slice(2, 10); let highlighted; try { highlighted = hljs.highlightAuto(code).value; } catch (err) { highlighted = code .replace(/&/g, '&') .replace(//g, '>'); } html += `
${lang} 
            
${highlighted}
`; } else { // TEXTO NORMAL let escaped = part .replace(/&/g, '&') .replace(//g, '>'); // Código inline: `texto` escaped = escaped.replace( /`([^`\n]+)`/g, '$1' ); // Quebras de linha escaped = escaped.replace(/\n/g, '
'); html += escaped; } } return html; }
