function parseAndRender(text) { if (!text) return ''; let html = ''; // Expressão regular corrigida para capturar blocos de código const parts = text.split(/(```[\s\S]*?```)/); for (let i = 0; i < parts.length; i++) { const part = parts[i]; if (part.startsWith('```') && part.endsWith('```')) { const content = part.slice(3, -3); const firstLineIndex = content.indexOf('\n'); let lang = 'code'; let code = content; if (firstLineIndex > 0) { const possibleLang = content.substring(0, firstLineIndex).trim(); if (/^[a-zA-Z]+$/.test(possibleLang)) { lang = possibleLang; code = content.substring(firstLineIndex + 1); } } const id = 'cb_' + Math.random().toString(36).slice(2, 10); const highlighted = hljs.highlightAuto(code).value; html += '
'; html += '
'; html += '' + lang.toUpperCase() + ''; html += ''; html += '
'; html += '
' + highlighted + '
'; html += '
'; } else { const escaped = part.replace(/&/g, '&').replace(//g, '>'); const withInlineCode = escaped.replace(/`([^`]+)`/g, '$1'); const withBreaks = withInlineCode.replace(/\n/g, '
'); html += withBreaks; } } return html; }
